//This program is to print out the yearly blance of a savings account but increase decimals in C++.

#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
int year;
float num;
float newnum = 0.00;
cout <<"Enter the starting amount: ";
cin >> num;
newnum = num;
year = 0;
cout <<"\n";
while (newnum <= num * 2)
	{
	year++;
	newnum = newnum * 1.1;
	cout << setprecision(year) << fixed;
	cout <<"Year\t"<< year <<"\t" << newnum <<"\n";
	}
return 0;
}
